### 7.3 Hook de Penalidades

```typescript
// hooks/usePenalties.ts

export function usePenalties(filters: PenaltyFilters) {
  const [penalties, setPenalties] = useState<HCPenalty[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  
  const supabase = createClient();

  useEffect(() => {
    fetchPenalties();
  }, [filters]);

  const fetchPenalties = async () => {
    setLoading(true);
    let query = supabase
      .from('hc_penalties')
      .select(`
        *,
        penalty_type:penalty_types(*),
        user:usuarios(id, nombres, apellidos, codigo_entel),
        store:tiendas(id, nombre, codigo)
      `);
    
    if (filters.year) query = query.eq('year', filters.year);
    if (filters.month) query = query.eq('month', filters.month);
    if (filters.userId) query = query.eq('user_id', filters.userId);
    if (filters.status) query = query.eq('status', filters.status);
    if (filters.penaltyTypeId) query = query.eq('penalty_type_id', filters.penaltyTypeId);
    
    const { data, error } = await query.order('created_at', { ascending: false });
    
    if (error) {
      setError(error.message);
    } else {
      setPenalties(data || []);
    }
    setLoading(false);
  };

  const waivePenalty = async (penaltyId: string, reason: string) => {
    const { error } = await supabase
      .from('hc_penalties')
      .update({
        status: 'waived',
        waived_reason: reason,
        waived_by: (await supabase.auth.getUser()).data.user?.id,
        waived_at: new Date().toISOString()
      })
      .eq('id', penaltyId);
    
    if (!error) {
      fetchPenalties();
    }
    return { error };
  };

  return { penalties, loading, error, refetch: fetchPenalties, waivePenalty };
}
```

### 7.4 Parser de Excel para FICHA

```typescript
// lib/penalidades/importParser.ts

import * as XLSX from 'xlsx';

interface ParsedRow {
  userCode: string;
  penaltyCode: string;
  quantity: number;
  originalAmount: number;
  incidentDate?: string;
  reference?: string;
}

interface ParseResult {
  rows: ParsedRow[];
  errors: ImportError[];
  warnings: ImportWarning[];
}

export async function parseFichaExcel(
  file: File,
  columnMapping: Record<string, string>
): Promise<ParseResult> {
  const buffer = await file.arrayBuffer();
  const workbook = XLSX.read(buffer, { type: 'array' });
  const sheet = workbook.Sheets[workbook.SheetNames[0]];
  const rawData = XLSX.utils.sheet_to_json(sheet, { header: 1 });
  
  const rows: ParsedRow[] = [];
  const errors: ImportError[] = [];
  const warnings: ImportWarning[] = [];
  
  // Saltar header
  for (let i = 1; i < rawData.length; i++) {
    const row = rawData[i] as any[];
    
    try {
      const parsed: ParsedRow = {
        userCode: String(row[columnMapping.userCode] || '').trim(),
        penaltyCode: String(row[columnMapping.penaltyCode] || '').trim(),
        quantity: Number(row[columnMapping.quantity]) || 1,
        originalAmount: Number(row[columnMapping.originalAmount]) || 0,
        incidentDate: parseDate(row[columnMapping.incidentDate]),
        reference: row[columnMapping.reference]
      };
      
      // Validaciones
      if (!parsed.userCode) {
        errors.push({ row: i + 1, field: 'userCode', message: 'Usuario vacÃ­o' });
        continue;
      }
      
      if (!parsed.penaltyCode) {
        errors.push({ row: i + 1, field: 'penaltyCode', message: 'Tipo de penalidad vacÃ­o' });
        continue;
      }
      
      rows.push(parsed);
    } catch (e) {
      errors.push({ row: i + 1, message: `Error procesando fila: ${e}` });
    }
  }
  
  return { rows, errors, warnings };
}
```
